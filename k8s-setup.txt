# Install Docker, you can ignore the warning from Docker about using WSL
curl -fsSL https://get.docker.com -o get-docker.sh
sudo sh get-docker.sh

# Add your user to the Docker group
sudo usermod -aG docker $USER

# Sanity check that both tools were installed successfully
docker --version
docker compose version

# First we're going to install kind as our kubernetes host environment:
[ $(uname -m) = x86_64 ] && curl -Lo ./kind https://kind.sigs.k8s.io/dl/v0.31.0/kind-linux-amd64
chmod +x ./kind
sudo mv ./kind /usr/local/bin/kind

# Now we're going to ask kind to create a k8s cluster with the name k8s-playground.
# The cluster will have a control-plane node and two worker nodes - one for frontend services and one for backend services.
# The kind configuration for this setup is contained in the kind-config.yaml file
kind create cluster --config ./kind-config.yaml

# At this point, you should be able to run the following command and see three docker containers running the image kindest/node:v....:
docker ps

# Now we're going to install kubectl - the command line tool for interacting with a k8s cluster
# Download installer and checksum file.
curl -LO "https://dl.k8s.io/release/$(curl -L -s https://dl.k8s.io/release/stable.txt)/bin/linux/amd64/kubectl"
curl -LO "https://dl.k8s.io/release/$(curl -L -s https://dl.k8s.io/release/stable.txt)/bin/linux/amd64/kubectl.sha256"

# Run checksum comparison - you should get a message saying kubectl: OK. If not, wait a while and start this bit again - the file's not right.
echo "$(cat kubectl.sha256)  kubectl" | sha256sum --check

# Run installer
sudo install -o root -g root -m 0755 kubectl /usr/local/bin/kubectl

# You should also now be able to run the kubectl command (the k8s cli tool) and see some default k8s pods that are set up by kind automatically:
kubectl get pods --all-namespaces

# We'll install helm in order to help us get some third party tools added to our k8s cluster
curl -fsSL -o get_helm.sh https://raw.githubusercontent.com/helm/helm/main/scripts/get-helm-4
chmod 700 get_helm.sh
./get_helm.sh
